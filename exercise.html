Palindrome Number

#include<stdio.h>
int main()
{
 	int n,r=0,temp,rev;
 	
	printf("Enter Number :");
 	scanf("%d",&n);

 	temp = n;
 	
 	while(n > 0)
 	{
   		r = n%10;
   		rev = rev * 10 + r;
   		n = n / 10;
 	}

  	if(temp == rev)
  	{
  		 printf("%d is Palindrome Number ",temp);
 	}
 	else
 	{
 		 printf("%d is Not Palindrome Number ",temp);
 	}

}

krishnamurthy Number

#include<stdio.h>
main()
{
	int n,i,temp=0,r,f=1,sum=0;
	
	printf("Enter number :");
	scanf("%d",&n);
	
	temp=n;
	while(n!=0)
	{
		
		r=n%10;
		f=1;
		
		for(i=1; i<=r; i++)
		{
			f=f*i;
		}
		
		sum=sum+f;
		n=n/10;
	}
	
		
	if(sum==temp)
	{
		printf("==> %d is Krishnamurthy Number <==",temp);
	}
	else
	{
		printf("==> %d is Not Krishnamurthy Number <==",temp);
	}
	
}
