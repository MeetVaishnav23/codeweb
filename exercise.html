Palindrome Number

#include <stdio.h>
main()
{
 int n,r=0,temp,rev;
 printf("Enter Number :");
 scanf("%d",&n);

 temp = n;
 while(n > 0)
 {
   r = n%10;
   rev = rev * 10 + r;
   n = n / 10;
  }

  if(temp == rev)
  {
   printf("%d is Palindrome Number ",temp);
 }
 else
 {
  printf("%d is Not Palindrome Number ",temp);
  }

}
